434 template<typename Index = int>
435 void run(Status<Index>& status, Float* Y) {
436 int& iter = status.iter;
437 Float multiplier = (iter < stop_lying_iter ? exaggeration_factor : 1);
438 Float momentum = (iter < mom_switch_iter ? start_momentum : final_momentum);
439
440 for(; iter < max_iter; ++iter) {
441 // Stop lying about the P-values after a while, and switch momentum
442 if (iter == stop_lying_iter) {
443 multiplier = 1;
444 }
445 if (iter == mom_switch_iter) {
446 momentum = final_momentum;
447 }
448
449 iterate(status, Y, multiplier, momentum);
450 }
451
452 return;
453 }

462 template<typename Index>
463 void iterate(Status<Index>& status, Float* Y, Float multiplier, Float momentum) {
464 compute_gradient(status, Y, multiplier);
465
466 auto& gains = status.gains;
467 auto& dY = status.dY;
468 auto& uY = status.uY;
469 auto& col_P = status.neighbors;
470
471 // Update gains
472 for (size_t i = 0; i < gains.size(); ++i) {
473 Float& g = gains[i];
474 constexpr Float lower_bound = 0.01;
475 constexpr Float to_add = 0.2;
476 constexpr Float to_mult = 0.8;
477 g = std::max(lower_bound, sign(dY[i]) != sign(uY[i]) ? (g + to_add) : (g * to_mult));
478 }
479
480 // Perform gradient update (with momentum and gains)
481 for (size_t i = 0; i < gains.size(); ++i) {
482 uY[i] = momentum * uY[i] - eta * gains[i] * dY[i];
483 Y[i] += uY[i];
484 }
485
486 // Make solution zero-mean
487 for (int d = 0; d < ndim; ++d) {
488 auto start = Y + d;
489 size_t N = col_P.size();
490
491 // Compute means from column-major coordinates.
492 Float sum = 0;
493 for (size_t i = 0; i < N; ++i, start += ndim) {
494 sum += *start;
495 }
496 sum /= N;
497
498 start = Y + d;
499 for (size_t i = 0; i < N; ++i, start += ndim) {
500 *start -= sum;
501 }
502 }
503
504 return;
505 }

508 template<typename Index>
509 void compute_gradient(Status<Index>& status, const Float* Y, Float multiplier) {
510 auto& tree = status.tree;
511 tree.set(Y);
512
513 compute_edge_forces(status, Y, multiplier);
514
515 size_t N = status.neighbors.size();
516 auto& neg_f = status.neg_f;
517 std::fill(neg_f.begin(), neg_f.end(), 0);
518
519 Float sum_Q = 0;
520 if (interpolation) {
521 sum_Q = interpolate::compute_non_edge_forces(
522 tree,
523 N,
524 Y,
525 theta,
526 neg_f.data(),
527 interpolation
528#ifdef _OPENMP
529 , status.omp_buffer
530#endif
531 );
532
533 } else {
534#ifdef _OPENMP
535 // Don't use reduction methods to ensure that we sum in a consistent order.
536 #pragma omp parallel for
537 for (size_t n = 0; n < N; ++n) {
538 status.omp_buffer[n] = status.tree.compute_non_edge_forces(n, theta, neg_f.data() + n * ndim);
539 }
540 sum_Q = std::accumulate(status.omp_buffer.begin(), status.omp_buffer.end(), static_cast<Float>(0));
541#else
542 for (size_t n = 0; n < N; ++n) {
543 sum_Q += status.tree.compute_non_edge_forces(n, theta, neg_f.data() + n * ndim);
544 }
545#endif
546 }
547
548 // Compute final t-SNE gradient
549 for (size_t i = 0; i < N * ndim; ++i) {
550 status.dY[i] = status.pos_f[i] - (neg_f[i] / sum_Q);
551 }
552 }

554 template<typename Index>
555 void compute_edge_forces(Status<Index>& status, const Float* Y, Float multiplier) {
556 const auto& neighbors = status.neighbors;
557 auto& pos_f = status.pos_f;
558 std::fill(pos_f.begin(), pos_f.end(), 0);
559
560 #pragma omp parallel for
561 for (size_t n = 0; n < neighbors.size(); ++n) {
562 const auto& current = neighbors[n];
563 const Float* self = Y + n * ndim;
564 Float* pos_out = pos_f.data() + n * ndim;
565
566 for (const auto& x : current) {
567 Float sqdist = 0;
568 const Float* neighbor = Y + x.first * ndim;
569 for (int d = 0; d < ndim; ++d) {
570 sqdist += (self[d] - neighbor[d]) * (self[d] - neighbor[d]);
571 }
572
573 const Float mult = multiplier * x.second / (static_cast<Float>(1) + sqdist);
574 for (int d = 0; d < ndim; ++d) {
575 pos_out[d] += mult * (self[d] - neighbor[d]);
576 }
577 }
578 }
579
580 return;
581 }

638 template<class Algorithm>
639 auto initialize(const Algorithm* searcher) {
640 const int K = perplexity_to_k(perplexity);
641 size_t N = searcher->nobs();
642 if (K >= N) {
643 throw std::runtime_error("number of observations should be greater than 3 * perplexity");
644 }
645
646#ifdef PROGRESS_PRINTER
647 PROGRESS_PRINTER("qdtsne::Tsne::initialize", "Searching for nearest neighbors")
648#endif
649 NeighborList<decltype(searcher->nobs()), Float> neighbors(N);
650
651 #pragma omp parallel for
652 for (size_t i = 0; i < N; ++i) {
653 neighbors[i] = searcher->find_nearest_neighbors(i, K);
654 }
655
656 return initialize_internal(std::move(neighbors), perplexity);
657 }

335  template<typename Index, typename Dist>
336  void compute_gaussian_perplexity(const std::vector<Dist*>& nn_dist, int K, Status<Index>& status) {
337  const size_t N = nn_dist.size();
338  constexpr double max_value = std::numeric_limits<double>::max();
339  constexpr double tol = 1e-5;
340  const double log_perplexity = std::log(static_cast<double>(K)/3.0); // implicitly taken from choice of 'k'.
341 
342  std::vector<std::vector<double> >& val_P = status.probabilities;
343 #ifdef _OPENMP
344  val_P.resize(N, std::vector<double>(K));
345 #endif
346 
347  #pragma omp parallel for
348  for (size_t n = 0; n < N; ++n){
349  double beta = 1.0;
350  double min_beta = 0, max_beta = max_value;
351  double sum_P = 0;
352  const double* distances = nn_dist[n];
353 
354 #ifdef _OPENMP
355  auto& output = val_P[n];
356 #else
357  val_P.emplace_back(K);
358  auto& output = val_P.back();
359 #endif
360 
361  // Iterate until we found a good perplexity
362  for (int iter = 0; iter < 200; ++iter) {
363  for (int m = 0; m < K; ++m) {
364  output[m] = std::exp(-beta * distances[m] * distances[m]); // apply gaussian kernel
365  }
366 
367  // Compute entropy of current row
368  sum_P = std::accumulate(output.begin(), output.end(), std::numeric_limits<double>::min());
369  double H = .0;
370  for(int m = 0; m < K; ++m) {
371  H += beta * distances[m] * distances[m] * output[m];
372  }
373  H = (H / sum_P) + std::log(sum_P);
374 
375  // Evaluate whether the entropy is within the tolerance level
376  double Hdiff = H - log_perplexity;
377  if (Hdiff < tol && -Hdiff < tol) {
378  break;
379  } else {
380  if (Hdiff > 0) {
381  min_beta = beta;
382  if (max_beta == max_value) {
383  beta *= 2.0;
384  } else {
385  beta = (beta + max_beta) / 2.0;
386  }
387  } else {
388  max_beta = beta;
389  beta = (beta + min_beta) / 2.0;
390  }
391  }
392  }
393 
394  // Row-normalize current row of P.
395  for (auto& o : output) {
396  o /= sum_P;
397  }
398  }
399 
400  return;
401  }

404  template<typename Index>
405  void symmetrize_matrix(const std::vector<Index*>& nn_index, int K, Status<typename std::remove_const<Index>::type>& status) {
406  auto& col_P = status.neighbors;
407  auto& probabilities = status.probabilities;
408  const size_t N = nn_index.size();
409 
410  // Initializing the output neighbor list.
411  for (size_t n = 0; n < N; ++n) {
412  col_P.emplace_back(nn_index[n], nn_index[n] + K);
413  }
414 
415  for (size_t n = 0; n < N; ++n) {
416  auto my_neighbors = nn_index[n];
417 
418  for (int k1 = 0; k1 < K; ++k1) {
419  auto curneighbor = my_neighbors[k1];
420  auto neighbors_neighbors = nn_index[curneighbor];
421 
422  // Check whether the current point is present in its neighbor's set.
423  bool present = false;
424  for (int k2 = 0; k2 < K; ++k2) {
425  if (neighbors_neighbors[k2] == n) {
426  if (n < curneighbor) {
427  // Adding the probabilities - but if n >= curneighbor, then this would have
428  // already been done at n = curneighbor, so we skip this to avoid adding it twice.
429  double sum = probabilities[n][k1] + probabilities[curneighbor][k2];
430  probabilities[n][k1] = sum;
431  probabilities[curneighbor][k2] = sum;
432  }
433  present = true;
434  break;
435  }
436  }
437 
438  if (!present) {
439  // If not present, no addition of probabilities is involved.
440  col_P[curneighbor].push_back(n);
441  probabilities[curneighbor].push_back(probabilities[n][k1]);
442  }
443  }
444  }
445 
446  // Divide the result by two
447  double total = 0;
448  for (auto& x : probabilities) {
449  for (auto& y : x) {
450  y /= 2;
451  total += y;
452  }
453  }
454 
455  // Probabilities across the entire matrix sum to unity.
456  for (auto& x : probabilities) {
457  for (auto& y : x) {
458  y /= total;
459  }
460  }
461 
462  return;
463  }

502  template<typename Index = int>
503  void run(Status<Index>& status, double* Y) {
504  int& iter = status.iter;
505  double multiplier = (iter < stop_lying_iter ? exaggeration_factor : 1);
506  double momentum = (iter < mom_switch_iter ? start_momentum : final_momentum);
507 
508  for(; iter < max_iter; ++iter) {
509  // Stop lying about the P-values after a while, and switch momentum
510  if (iter == stop_lying_iter) {
511  multiplier = 1;
512  }
513  if (iter == mom_switch_iter) {
514  momentum = final_momentum;
515  }
516 
517  iterate(status, Y, multiplier, momentum);
518  }
519 
520  return;
521  }

528  template<typename Index>
529  void iterate(Status<Index>& status, double* Y, double multiplier, double momentum) {
530  compute_gradient(status, Y, multiplier);
531 
532  auto& gains = status.gains;
533  auto& dY = status.dY;
534  auto& uY = status.uY;
535  auto& col_P = status.neighbors;
536 
537  // Update gains
538  for (size_t i = 0; i < gains.size(); ++i) {
539  double& g = gains[i];
540  g = std::max(0.01, sign(dY[i]) != sign(uY[i]) ? (g + 0.2) : (g * 0.8));
541  }
542 
543  // Perform gradient update (with momentum and gains)
544  for (size_t i = 0; i < gains.size(); ++i) {
545  uY[i] = momentum * uY[i] - eta * gains[i] * dY[i];
546  Y[i] += uY[i];
547  }
548 
549  // Make solution zero-mean
550  for (int d = 0; d < ndim; ++d) {
551  auto start = Y + d;
552  size_t N = col_P.size();
553 
554  // Compute means from column-major coordinates.
555  double sum = 0;
556  for (size_t i = 0; i < N; ++i, start += ndim) {
557  sum += *start;
558  }
559  sum /= N;
560 
561  start = Y + d;
562  for (size_t i = 0; i < N; ++i, start += ndim) {
563  *start -= sum;
564  }
565  }
566 
567  return;
568  }

571  template<typename Index>
572  void compute_gradient(Status<Index>& status, const double* Y, double multiplier) {
573  auto& tree = status.tree;
574  tree.set(Y);
575 
576  compute_edge_forces(status, Y, multiplier);
577 
578  size_t N = status.neighbors.size();
579  auto& neg_f = status.neg_f;
580  std::fill(neg_f.begin(), neg_f.end(), 0);
581 
582 #ifdef _OPENMP
583  // Don't use reduction methods to ensure that we sum in a consistent order.
584  #pragma omp parallel for
585  for (size_t n = 0; n < N; ++n) {
586  status.omp_buffer[n] = status.tree.compute_non_edge_forces(n, theta, neg_f.data() + n * ndim);
587  }
588  double sum_Q = std::accumulate(status.omp_buffer.begin(), status.omp_buffer.end(), 0.0);
589 #else
590  double sum_Q = 0;
591  for (size_t n = 0; n < N; ++n) {
592  sum_Q += status.tree.compute_non_edge_forces(n, theta, neg_f.data() + n * ndim);
593  }
594 #endif
595 
596  // Compute final t-SNE gradient
597  for (size_t i = 0; i < N * ndim; ++i) {
598  status.dY[i] = status.pos_f[i] - (neg_f[i] / sum_Q);
599  }
600  }

602  template<typename Index>
603  void compute_edge_forces(Status<Index>& status, const double* Y, double multiplier) {
604  const auto& col_P = status.neighbors;
605  const auto& val_P = status.probabilities;
606  auto& pos_f = status.pos_f;
607  std::fill(pos_f.begin(), pos_f.end(), 0);
608 
609  #pragma omp parallel for
610  for (size_t n = 0; n < col_P.size(); ++n) {
611  const auto& cur_prob = val_P[n];
612  const auto& cur_col = col_P[n];
613  const double* self = Y + n * ndim;
614  double* pos_out = pos_f.data() + n * ndim;
615 
616  for (size_t i = 0; i < cur_col.size(); ++i) {
617  double sqdist = 0;
618  const double* neighbor = Y + cur_col[i] * ndim;
619  for (int d = 0; d < ndim; ++d) {
620  sqdist += (self[d] - neighbor[d]) * (self[d] - neighbor[d]);
621  }
622 
623  const double mult = multiplier * cur_prob[i] / (1 + sqdist);
624  for (int d = 0; d < ndim; ++d) {
625  pos_out[d] += mult * (self[d] - neighbor[d]);
626  }
627  }
628  }
629 
630  return;
631  }

648  template<class Algorithm = knncolle::VpTreeEuclidean<>, typename Input = double>
649  auto initialize(const Input* input, size_t D, size_t N) {
650  Algorithm searcher(D, N, input);
651  const int K = std::ceil(perplexity * 3);
652  if (K >= N) {
653  throw std::runtime_error("number of observations should be greater than 3 * perplexity");
654  }
655 
656  std::vector<int> indices(N * K);
657  std::vector<double> distances(N * K);
658  std::vector<const int*> nn_index(N);
659  std::vector<const double*> nn_dist(N);
660 
661  for (size_t i = 0; i < N; ++i) {
662  auto out = searcher.find_nearest_neighbors(i, K);
663  for (size_t k = 0; k < out.size(); ++k) {
664  indices[k + i *K] = out[k].first;
665  distances[k + i *K] = out[k].second;
666  }
667  nn_index[i] = indices.data() + i * K;
668  nn_dist[i] = distances.data() + i * K;
669  }
670 
671  return initialize(nn_index, nn_dist, K);
672  }

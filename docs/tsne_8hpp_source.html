339 template<typename Index, typename Dist>
340 void compute_gaussian_perplexity(const std::vector<Dist*>& nn_dist, int K, Status<Index>& status) {
341 const size_t N = nn_dist.size();
342 constexpr double max_value = std::numeric_limits<double>::max();
343 constexpr double tol = 1e-5;
344 const double log_perplexity = std::log(static_cast<double>(K)/3.0); // implicitly taken from choice of 'k'.
345
346 std::vector<std::vector<double> >& val_P = status.probabilities;
347#ifdef _OPENMP
348 val_P.resize(N, std::vector<double>(K));
349#endif
350
351 #pragma omp parallel
352 {
353 std::vector<double> squared_delta_dist(K);
354 std::vector<double> quad_delta_dist(K);
355
356 #pragma omp for
357 for (size_t n = 0; n < N; ++n){
358 double beta = 1.0;
359 double min_beta = 0, max_beta = max_value;
360 double sum_P = 0;
361 const double* distances = nn_dist[n];
362
363#ifdef _OPENMP
364 auto& output = val_P[n];
365#else
366 val_P.emplace_back(K);
367 auto& output = val_P.back();
368#endif
369
370 // We adjust the probabilities by subtracting the first squared
371 // distance from everything. This avoids problems with underflow
372 // when converting distances to probabilities; it otherwise has no
373 // effect on the entropy or even the final probabilities because it
374 // just scales all probabilities up/down (and they need to be
375 // normalized anyway, so any scaling effect just cancels out).
376 const double first = distances[0] * distances[0];
377 for (int m = 1; m < K; ++m) {
378 squared_delta_dist[m] = distances[m] * distances[m] - first;
379 quad_delta_dist[m] = squared_delta_dist[m] * squared_delta_dist[m];
380 }
381 output[0] = 1;
382
383 for (int iter = 0; iter < 200; ++iter) {
384 // Apply gaussian kernel. We skip the first value because
385 // we effectively normalized it to 1 by subtracting 'first'.
386 for (int m = 1; m < K; ++m) {
387 output[m] = std::exp(-beta * squared_delta_dist[m]);
388 }
389
390 sum_P = std::accumulate(output.begin() + 1, output.end(), 1.0);
391 const double prod = std::inner_product(squared_delta_dist.begin() + 1, squared_delta_dist.end(), output.begin() + 1, 0.0);
392 const double entropy = beta * (prod / sum_P) + std::log(sum_P);
393
394 const double diff = entropy - log_perplexity;
395 if (std::abs(diff) < tol) {
396 break;
397 }
398
399 // Attempt a Newton-Raphson search first.
400 bool nr_ok = false;
401#ifndef QDTSNE_BETA_BINARY_SEARCH_ONLY
402 const double prod2 = std::inner_product(quad_delta_dist.begin() + 1, quad_delta_dist.end(), output.begin() + 1, 0.0);
403 const double d1 = - beta / sum_P * (prod2 - prod * prod / sum_P);
404 if (d1) {
405 const double alt_beta = beta - (diff / d1); // if it overflows, we should get Inf or -Inf, so the following comparison should be fine.
406 if (alt_beta > min_beta && alt_beta < max_beta) {
407 beta = alt_beta;
408 nr_ok = true;
409 }
410 }
411#endif
412
413 // Otherwise do a binary search.
414 if (!nr_ok) {
415 if (diff > 0) {
416 min_beta = beta;
417 if (max_beta == max_value) {
418 beta *= 2.0;
419 } else {
420 beta = (beta + max_beta) / 2.0;
421 }
422 } else {
423 max_beta = beta;
424 beta = (beta + min_beta) / 2.0;
425 }
426 }
427 }
428
429 // Row-normalize current row of P.
430 for (auto& o : output) {
431 o /= sum_P;
432 }
433 }
434 }
435
436 return;
437 }

440 template<typename Index>
441 void symmetrize_matrix(const std::vector<Index*>& nn_index, int K, Status<typename std::remove_const<Index>::type>& status) {
442 auto& col_P = status.neighbors;
443 auto& probabilities = status.probabilities;
444 const size_t N = nn_index.size();
445
446 // Initializing the output neighbor list.
447 for (size_t n = 0; n < N; ++n) {
448 col_P.emplace_back(nn_index[n], nn_index[n] + K);
449 }
450
451 for (size_t n = 0; n < N; ++n) {
452 auto my_neighbors = nn_index[n];
453
454 for (int k1 = 0; k1 < K; ++k1) {
455 auto curneighbor = my_neighbors[k1];
456 auto neighbors_neighbors = nn_index[curneighbor];
457
458 // Check whether the current point is present in its neighbor's set.
459 bool present = false;
460 for (int k2 = 0; k2 < K; ++k2) {
461 if (neighbors_neighbors[k2] == n) {
462 if (n < curneighbor) {
463 // Adding the probabilities - but if n >= curneighbor, then this would have
464 // already been done at n = curneighbor, so we skip this to avoid adding it twice.
465 double sum = probabilities[n][k1] + probabilities[curneighbor][k2];
466 probabilities[n][k1] = sum;
467 probabilities[curneighbor][k2] = sum;
468 }
469 present = true;
470 break;
471 }
472 }
473
474 if (!present) {
475 // If not present, no addition of probabilities is involved.
476 col_P[curneighbor].push_back(n);
477 probabilities[curneighbor].push_back(probabilities[n][k1]);
478 }
479 }
480 }
481
482 // Divide the result by two
483 double total = 0;
484 for (auto& x : probabilities) {
485 for (auto& y : x) {
486 y /= 2;
487 total += y;
488 }
489 }
490
491 // Probabilities across the entire matrix sum to unity.
492 for (auto& x : probabilities) {
493 for (auto& y : x) {
494 y /= total;
495 }
496 }
497
498 return;
499 }

538 template<typename Index = int>
539 void run(Status<Index>& status, double* Y) {
540 int& iter = status.iter;
541 double multiplier = (iter < stop_lying_iter ? exaggeration_factor : 1);
542 double momentum = (iter < mom_switch_iter ? start_momentum : final_momentum);
543
544 for(; iter < max_iter; ++iter) {
545 // Stop lying about the P-values after a while, and switch momentum
546 if (iter == stop_lying_iter) {
547 multiplier = 1;
548 }
549 if (iter == mom_switch_iter) {
550 momentum = final_momentum;
551 }
552
553 iterate(status, Y, multiplier, momentum);
554 }
555
556 return;
557 }

561 return (x == .0 ? .0 : (x < .0 ? -1.0 : 1.0));

564 template<typename Index>
565 void iterate(Status<Index>& status, double* Y, double multiplier, double momentum) {
566 compute_gradient(status, Y, multiplier);
567
568 auto& gains = status.gains;
569 auto& dY = status.dY;
570 auto& uY = status.uY;
571 auto& col_P = status.neighbors;
572
573 // Update gains
574 for (size_t i = 0; i < gains.size(); ++i) {
575 double& g = gains[i];
576 g = std::max(0.01, sign(dY[i]) != sign(uY[i]) ? (g + 0.2) : (g * 0.8));
577 }
578
579 // Perform gradient update (with momentum and gains)
580 for (size_t i = 0; i < gains.size(); ++i) {
581 uY[i] = momentum * uY[i] - eta * gains[i] * dY[i];
582 Y[i] += uY[i];
583 }
584
585 // Make solution zero-mean
586 for (int d = 0; d < ndim; ++d) {
587 auto start = Y + d;
588 size_t N = col_P.size();
589
590 // Compute means from column-major coordinates.
591 double sum = 0;
592 for (size_t i = 0; i < N; ++i, start += ndim) {
593 sum += *start;
594 }
595 sum /= N;
596
597 start = Y + d;
598 for (size_t i = 0; i < N; ++i, start += ndim) {
599 *start -= sum;
600 }
601 }
602
603 return;
604 }

607 template<typename Index>
608 void compute_gradient(Status<Index>& status, const double* Y, double multiplier) {
609 auto& tree = status.tree;
610 tree.set(Y);
611
612 compute_edge_forces(status, Y, multiplier);
613
614 size_t N = status.neighbors.size();
615 auto& neg_f = status.neg_f;
616 std::fill(neg_f.begin(), neg_f.end(), 0);
617
618#ifdef _OPENMP
619 // Don't use reduction methods to ensure that we sum in a consistent order.
620 #pragma omp parallel for
621 for (size_t n = 0; n < N; ++n) {
622 status.omp_buffer[n] = status.tree.compute_non_edge_forces(n, theta, neg_f.data() + n * ndim);
623 }
624 double sum_Q = std::accumulate(status.omp_buffer.begin(), status.omp_buffer.end(), 0.0);
625#else
626 double sum_Q = 0;
627 for (size_t n = 0; n < N; ++n) {
628 sum_Q += status.tree.compute_non_edge_forces(n, theta, neg_f.data() + n * ndim);
629 }
630#endif
631
632 // Compute final t-SNE gradient
633 for (size_t i = 0; i < N * ndim; ++i) {
634 status.dY[i] = status.pos_f[i] - (neg_f[i] / sum_Q);
635 }
636 }

638 template<typename Index>
639 void compute_edge_forces(Status<Index>& status, const double* Y, double multiplier) {
640 const auto& col_P = status.neighbors;
641 const auto& val_P = status.probabilities;
642 auto& pos_f = status.pos_f;
643 std::fill(pos_f.begin(), pos_f.end(), 0);
644
645 #pragma omp parallel for
646 for (size_t n = 0; n < col_P.size(); ++n) {
647 const auto& cur_prob = val_P[n];
648 const auto& cur_col = col_P[n];
649 const double* self = Y + n * ndim;
650 double* pos_out = pos_f.data() + n * ndim;
651
652 for (size_t i = 0; i < cur_col.size(); ++i) {
653 double sqdist = 0;
654 const double* neighbor = Y + cur_col[i] * ndim;
655 for (int d = 0; d < ndim; ++d) {
656 sqdist += (self[d] - neighbor[d]) * (self[d] - neighbor[d]);
657 }
658
659 const double mult = multiplier * cur_prob[i] / (1 + sqdist);
660 for (int d = 0; d < ndim; ++d) {
661 pos_out[d] += mult * (self[d] - neighbor[d]);
662 }
663 }
664 }
665
666 return;
667 }

721 template<class Algorithm>
722 auto initialize(const Algorithm* searcher) {
723 const int K = std::ceil(perplexity * 3);
724 size_t N = searcher->nobs();
725 if (K >= N) {
726 throw std::runtime_error("number of observations should be greater than 3 * perplexity");
727 }
728
729 std::vector<int> indices(N * K);
730 std::vector<double> distances(N * K);
731 std::vector<const int*> nn_index(N);
732 std::vector<const double*> nn_dist(N);
733
734 #pragma omp parallel for
735 for (size_t i = 0; i < N; ++i) {
736 auto out = searcher->find_nearest_neighbors(i, K);
737 for (size_t k = 0; k < out.size(); ++k) {
738 indices[k + i *K] = out[k].first;
739 distances[k + i *K] = out[k].second;
740 }
741 nn_index[i] = indices.data() + i * K;
742 nn_dist[i] = distances.data() + i * K;
743 }
744
745 return initialize(nn_index, nn_dist, K);
746 }

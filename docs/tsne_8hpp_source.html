408 template<typename Index = int>
409 void run(Status<Index>& status, double* Y) {
410 int& iter = status.iter;
411 double multiplier = (iter < stop_lying_iter ? exaggeration_factor : 1);
412 double momentum = (iter < mom_switch_iter ? start_momentum : final_momentum);
413
414 for(; iter < max_iter; ++iter) {
415 // Stop lying about the P-values after a while, and switch momentum
416 if (iter == stop_lying_iter) {
417 multiplier = 1;
418 }
419 if (iter == mom_switch_iter) {
420 momentum = final_momentum;
421 }
422
423 iterate(status, Y, multiplier, momentum);
424 }
425
426 return;
427 }

434 template<typename Index>
435 void iterate(Status<Index>& status, double* Y, double multiplier, double momentum) {
436 compute_gradient(status, Y, multiplier);
437
438 auto& gains = status.gains;
439 auto& dY = status.dY;
440 auto& uY = status.uY;
441 auto& col_P = status.neighbors;
442
443 // Update gains
444 for (size_t i = 0; i < gains.size(); ++i) {
445 double& g = gains[i];
446 g = std::max(0.01, sign(dY[i]) != sign(uY[i]) ? (g + 0.2) : (g * 0.8));
447 }
448
449 // Perform gradient update (with momentum and gains)
450 for (size_t i = 0; i < gains.size(); ++i) {
451 uY[i] = momentum * uY[i] - eta * gains[i] * dY[i];
452 Y[i] += uY[i];
453 }
454
455 // Make solution zero-mean
456 for (int d = 0; d < ndim; ++d) {
457 auto start = Y + d;
458 size_t N = col_P.size();
459
460 // Compute means from column-major coordinates.
461 double sum = 0;
462 for (size_t i = 0; i < N; ++i, start += ndim) {
463 sum += *start;
464 }
465 sum /= N;
466
467 start = Y + d;
468 for (size_t i = 0; i < N; ++i, start += ndim) {
469 *start -= sum;
470 }
471 }
472
473 return;
474 }

477 template<typename Index>
478 void compute_gradient(Status<Index>& status, const double* Y, double multiplier) {
479 auto& tree = status.tree;
480 tree.set(Y);
481
482 compute_edge_forces(status, Y, multiplier);
483
484 size_t N = status.neighbors.size();
485 auto& neg_f = status.neg_f;
486 std::fill(neg_f.begin(), neg_f.end(), 0);
487
488#ifdef _OPENMP
489 // Don't use reduction methods to ensure that we sum in a consistent order.
490 #pragma omp parallel for
491 for (size_t n = 0; n < N; ++n) {
492 status.omp_buffer[n] = status.tree.compute_non_edge_forces(n, theta, neg_f.data() + n * ndim);
493 }
494 double sum_Q = std::accumulate(status.omp_buffer.begin(), status.omp_buffer.end(), 0.0);
495#else
496 double sum_Q = 0;
497 for (size_t n = 0; n < N; ++n) {
498 sum_Q += status.tree.compute_non_edge_forces(n, theta, neg_f.data() + n * ndim);
499 }
500#endif
501
502 // Compute final t-SNE gradient
503 for (size_t i = 0; i < N * ndim; ++i) {
504 status.dY[i] = status.pos_f[i] - (neg_f[i] / sum_Q);
505 }
506 }

508 template<typename Index>
509 void compute_edge_forces(Status<Index>& status, const double* Y, double multiplier) {
510 const auto& neighbors = status.neighbors;
511 auto& pos_f = status.pos_f;
512 std::fill(pos_f.begin(), pos_f.end(), 0);
513
514 #pragma omp parallel for
515 for (size_t n = 0; n < neighbors.size(); ++n) {
516 const auto& current = neighbors[n];
517 const double* self = Y + n * ndim;
518 double* pos_out = pos_f.data() + n * ndim;
519
520 for (const auto& x : current) {
521 double sqdist = 0;
522 const double* neighbor = Y + x.first * ndim;
523 for (int d = 0; d < ndim; ++d) {
524 sqdist += (self[d] - neighbor[d]) * (self[d] - neighbor[d]);
525 }
526
527 const double mult = multiplier * x.second / (1 + sqdist);
528 for (int d = 0; d < ndim; ++d) {
529 pos_out[d] += mult * (self[d] - neighbor[d]);
530 }
531 }
532 }
533
534 return;
535 }

592 template<class Algorithm>
593 auto initialize(const Algorithm* searcher) {
594 const int K = std::ceil(perplexity * 3);
595 size_t N = searcher->nobs();
596 if (K >= N) {
597 throw std::runtime_error("number of observations should be greater than 3 * perplexity");
598 }
599
600#ifdef PROGRESS_PRINTER
601 PROGRESS_PRINTER("qdtsne::Tsne::initialize", "Searching for nearest neighbors")
602#endif
603 NeighborList<decltype(searcher->nobs())> neighbors(N);
604
605 #pragma omp parallel for
606 for (size_t i = 0; i < N; ++i) {
607 neighbors[i] = searcher->find_nearest_neighbors(i, K);
608 }
609
610 return initialize_internal(std::move(neighbors), perplexity);
611 }

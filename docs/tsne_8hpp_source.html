429 template<typename Index = int>
430 void run(Status<Index>& status, double* Y) {
431 int& iter = status.iter;
432 double multiplier = (iter < stop_lying_iter ? exaggeration_factor : 1);
433 double momentum = (iter < mom_switch_iter ? start_momentum : final_momentum);
434
435 for(; iter < max_iter; ++iter) {
436 // Stop lying about the P-values after a while, and switch momentum
437 if (iter == stop_lying_iter) {
438 multiplier = 1;
439 }
440 if (iter == mom_switch_iter) {
441 momentum = final_momentum;
442 }
443
444 iterate(status, Y, multiplier, momentum);
445 }
446
447 return;
448 }

451 static double sign(double x) {
452 return (x == .0 ? .0 : (x < .0 ? -1.0 : 1.0));

455 template<typename Index>
456 void iterate(Status<Index>& status, double* Y, double multiplier, double momentum) {
457 compute_gradient(status, Y, multiplier);
458
459 auto& gains = status.gains;
460 auto& dY = status.dY;
461 auto& uY = status.uY;
462 auto& col_P = status.neighbors;
463
464 // Update gains
465 for (size_t i = 0; i < gains.size(); ++i) {
466 double& g = gains[i];
467 g = std::max(0.01, sign(dY[i]) != sign(uY[i]) ? (g + 0.2) : (g * 0.8));
468 }
469
470 // Perform gradient update (with momentum and gains)
471 for (size_t i = 0; i < gains.size(); ++i) {
472 uY[i] = momentum * uY[i] - eta * gains[i] * dY[i];
473 Y[i] += uY[i];
474 }
475
476 // Make solution zero-mean
477 for (int d = 0; d < ndim; ++d) {
478 auto start = Y + d;
479 size_t N = col_P.size();
480
481 // Compute means from column-major coordinates.
482 double sum = 0;
483 for (size_t i = 0; i < N; ++i, start += ndim) {
484 sum += *start;
485 }
486 sum /= N;
487
488 start = Y + d;
489 for (size_t i = 0; i < N; ++i, start += ndim) {
490 *start -= sum;
491 }
492 }
493
494 return;
495 }

497private:
498 template<typename Index>
499 void compute_gradient(Status<Index>& status, const double* Y, double multiplier) {
500 auto& tree = status.tree;
501 tree.set(Y);
502
503 compute_edge_forces(status, Y, multiplier);
504
505 size_t N = status.neighbors.size();
506 auto& neg_f = status.neg_f;
507 std::fill(neg_f.begin(), neg_f.end(), 0);
508
509 double sum_Q = 0;
510 if (interpolation) {
511 sum_Q = interpolate::compute_non_edge_forces(
512 tree,
513 N,
514 Y,
515 theta,
516 neg_f.data(),
517 interpolation
518#ifdef _OPENMP
519 , status.omp_buffer
520#endif
521 );
522
523 } else {
524#ifdef _OPENMP
525 // Don't use reduction methods to ensure that we sum in a consistent order.
526 #pragma omp parallel for
527 for (size_t n = 0; n < N; ++n) {
528 status.omp_buffer[n] = status.tree.compute_non_edge_forces(n, theta, neg_f.data() + n * ndim);
529 }
530 sum_Q = std::accumulate(status.omp_buffer.begin(), status.omp_buffer.end(), 0.0);
531#else
532 for (size_t n = 0; n < N; ++n) {
533 sum_Q += status.tree.compute_non_edge_forces(n, theta, neg_f.data() + n * ndim);
534 }
535#endif
536 }
537
538 // Compute final t-SNE gradient
539 for (size_t i = 0; i < N * ndim; ++i) {
540 status.dY[i] = status.pos_f[i] - (neg_f[i] / sum_Q);
541 }
542 }

544 template<typename Index>
545 void compute_edge_forces(Status<Index>& status, const double* Y, double multiplier) {
546 const auto& neighbors = status.neighbors;
547 auto& pos_f = status.pos_f;
548 std::fill(pos_f.begin(), pos_f.end(), 0);
549
550 #pragma omp parallel for
551 for (size_t n = 0; n < neighbors.size(); ++n) {
552 const auto& current = neighbors[n];
553 const double* self = Y + n * ndim;
554 double* pos_out = pos_f.data() + n * ndim;
555
556 for (const auto& x : current) {
557 double sqdist = 0;
558 const double* neighbor = Y + x.first * ndim;
559 for (int d = 0; d < ndim; ++d) {
560 sqdist += (self[d] - neighbor[d]) * (self[d] - neighbor[d]);
561 }
562
563 const double mult = multiplier * x.second / (1 + sqdist);
564 for (int d = 0; d < ndim; ++d) {
565 pos_out[d] += mult * (self[d] - neighbor[d]);
566 }
567 }
568 }
569
570 return;
571 }

628 template<class Algorithm>
629 auto initialize(const Algorithm* searcher) {
630 const int K = std::ceil(perplexity * 3);
631 size_t N = searcher->nobs();
632 if (K >= N) {
633 throw std::runtime_error("number of observations should be greater than 3 * perplexity");
634 }
635
636#ifdef PROGRESS_PRINTER
637 PROGRESS_PRINTER("qdtsne::Tsne::initialize", "Searching for nearest neighbors")
638#endif
639 NeighborList<decltype(searcher->nobs())> neighbors(N);
640
641 #pragma omp parallel for
642 for (size_t i = 0; i < N; ++i) {
643 neighbors[i] = searcher->find_nearest_neighbors(i, K);
644 }
645
646 return initialize_internal(std::move(neighbors), perplexity);
647 }

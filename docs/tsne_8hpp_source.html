435 void run(Float* Y, int limit) {
436 Float multiplier = (iter < iparams.stop_lying_iter ? iparams.exaggeration_factor : 1);
437 Float momentum = (iter < iparams.mom_switch_iter ? iparams.start_momentum : iparams.final_momentum);
438
439 for(; iter < limit; ++iter) {
440 // Stop lying about the P-values after a while, and switch momentum
441 if (iter == iparams.stop_lying_iter) {
442 multiplier = 1;
443 }
444 if (iter == iparams.mom_switch_iter) {
445 momentum = iparams.final_momentum;
446 }
447
448 iterate(Y, multiplier, momentum);
449 }
450 }

466 static Float sign(Float x) {
467 constexpr Float zero = 0;
468 constexpr Float one = 1;
469 return (x == zero ? zero : (x < zero ? -one : one));
470 }

472 void iterate(Float* Y, Float multiplier, Float momentum) {
473 compute_gradient(Y, multiplier);
474
475 // Update gains
476 for (size_t i = 0; i < gains.size(); ++i) {
477 Float& g = gains[i];
478 constexpr Float lower_bound = 0.01;
479 constexpr Float to_add = 0.2;
480 constexpr Float to_mult = 0.8;
481 g = std::max(lower_bound, sign(dY[i]) != sign(uY[i]) ? (g + to_add) : (g * to_mult));
482 }
483
484 // Perform gradient update (with momentum and gains)
485 for (size_t i = 0; i < gains.size(); ++i) {
486 uY[i] = momentum * uY[i] - iparams.eta * gains[i] * dY[i];
487 Y[i] += uY[i];
488 }
489
490 // Make solution zero-mean
491 size_t N = nobs();
492 for (int d = 0; d < ndim; ++d) {
493 auto start = Y + d;
494
495 // Compute means from column-major coordinates.
496 Float sum = 0;
497 for (size_t i = 0; i < N; ++i, start += ndim) {
498 sum += *start;
499 }
500 sum /= N;
501
502 start = Y + d;
503 for (size_t i = 0; i < N; ++i, start += ndim) {
504 *start -= sum;
505 }
506 }
507
508 return;
509 }

538 }

542 for (size_t n = 0; n < N; ++n) {
543 sum_Q += tree.compute_non_edge_forces(n, iparams.theta, neg_f.data() + n * ndim);
544 }

548 void compute_gradient(const Float* Y, Float multiplier) {
549 tree.set(Y);
550 compute_edge_forces(Y, multiplier);
551
552 size_t N = nobs();
553 std::fill(neg_f.begin(), neg_f.end(), 0);
554
555 Float sum_Q = 0;
556 if (iparams.interpolation) {
557 Interpolator<ndim, Float> inter;
558 inter.set_num_threads(iparams.nthreads);
559 sum_Q = inter.compute_non_edge_forces(
560 tree,
561 N,
562 Y,
563 iparams.theta,
564 neg_f.data(),
565 iparams.interpolation,
566 parallel_buffer
567 );
568
569 } else {
570 sum_Q = compute_non_edge_forces();
571 }
572
573 // Compute final t-SNE gradient
574 for (size_t i = 0; i < N * ndim; ++i) {
575 dY[i] = pos_f[i] - (neg_f[i] / sum_Q);
576 }
577 }
578
579 void compute_edge_forces(const Float* Y, Float multiplier) {
580 std::fill(pos_f.begin(), pos_f.end(), 0);
581
582#ifndef QDTSNE_CUSTOM_PARALLEL
583 #pragma omp parallel for num_threads(iparams.nthreads)
584 for (size_t n = 0; n < neighbors.size(); ++n) {
585#else
586 QDTSNE_CUSTOM_PARALLEL(neighbors.size(), [&](size_t first_, size_t last_) -> void {
587 for (size_t n = first_; n < last_; ++n) {
588#endif
589
590 const auto& current = neighbors[n];
591 const Float* self = Y + n * ndim;
592 Float* pos_out = pos_f.data() + n * ndim;
593
594 for (const auto& x : current) {
595 Float sqdist = 0;
596 const Float* neighbor = Y + x.first * ndim;
597 for (int d = 0; d < ndim; ++d) {
598 sqdist += (self[d] - neighbor[d]) * (self[d] - neighbor[d]);
599 }
600
601 const Float mult = multiplier * x.second / (static_cast<Float>(1) + sqdist);
602 for (int d = 0; d < ndim; ++d) {
603 pos_out[d] += mult * (self[d] - neighbor[d]);
604 }
605 }
606
607#ifndef QDTSNE_CUSTOM_PARALLEL
608 }
609#else
610 }
611 }, iparams.nthreads);
612#endif
613
614 return;
615 }

722 template<class Algorithm>
723 auto initialize(const Algorithm* searcher) {
724 const int K = perplexity_to_k(perplexity);
725 size_t N = searcher->nobs();
726 if (K >= N) {
727 throw std::runtime_error("number of observations should be greater than 3 * perplexity");
728 }
729
730 NeighborList<decltype(searcher->nobs()), Float> neighbors(N);

732#ifndef QDTSNE_CUSTOM_PARALLEL

735#else

737 for (size_t i = first_; i < last_; ++i) {
738#endif

741
742#ifndef QDTSNE_CUSTOM_PARALLEL
743 }
744#else

747#endif

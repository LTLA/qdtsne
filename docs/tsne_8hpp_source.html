480 template<typename Index = int>
481 void run(Status<Index>& status, Float* Y) {
482 int& iter = status.iter;
483 Float multiplier = (iter < stop_lying_iter ? exaggeration_factor : 1);
484 Float momentum = (iter < mom_switch_iter ? start_momentum : final_momentum);
485
486 for(; iter < max_iter; ++iter) {
487 // Stop lying about the P-values after a while, and switch momentum
488 if (iter == stop_lying_iter) {
489 multiplier = 1;
490 }
491 if (iter == mom_switch_iter) {
492 momentum = final_momentum;
493 }
494
495 iterate(status, Y, multiplier, momentum);
496 }
497
498 return;
499 }

508 template<typename Index>
509 void iterate(Status<Index>& status, Float* Y, Float multiplier, Float momentum) {
510 compute_gradient(status, Y, multiplier);
511
512 auto& gains = status.gains;
513 auto& dY = status.dY;
514 auto& uY = status.uY;
515 auto& col_P = status.neighbors;
516
517 // Update gains
518 for (size_t i = 0; i < gains.size(); ++i) {
519 Float& g = gains[i];
520 constexpr Float lower_bound = 0.01;
521 constexpr Float to_add = 0.2;
522 constexpr Float to_mult = 0.8;
523 g = std::max(lower_bound, sign(dY[i]) != sign(uY[i]) ? (g + to_add) : (g * to_mult));
524 }
525
526 // Perform gradient update (with momentum and gains)
527 for (size_t i = 0; i < gains.size(); ++i) {
528 uY[i] = momentum * uY[i] - eta * gains[i] * dY[i];
529 Y[i] += uY[i];
530 }
531
532 // Make solution zero-mean
533 for (int d = 0; d < ndim; ++d) {
534 auto start = Y + d;
535 size_t N = col_P.size();
536
537 // Compute means from column-major coordinates.
538 Float sum = 0;
539 for (size_t i = 0; i < N; ++i, start += ndim) {
540 sum += *start;
541 }
542 sum /= N;
543
544 start = Y + d;
545 for (size_t i = 0; i < N; ++i, start += ndim) {
546 *start -= sum;
547 }
548 }
549
550 return;
551 }

588 template<typename Index>
589 void compute_gradient(Status<Index>& status, const Float* Y, Float multiplier) {
590 auto& tree = status.tree;
591 tree.set(Y);
592
593 compute_edge_forces(status, Y, multiplier);
594
595 size_t N = status.neighbors.size();
596 auto& neg_f = status.neg_f;
597 std::fill(neg_f.begin(), neg_f.end(), 0);
598
599 Float sum_Q = 0;
600 if (interpolation) {
601 Interpolator<ndim, Float> inter;
602 inter.set_num_threads(nthreads);
603 sum_Q = inter.compute_non_edge_forces(
604 tree,
605 N,
606 Y,
607 theta,
608 neg_f.data(),
609 interpolation,
610 status.parallel_buffer
611 );
612
613 } else {
614 sum_Q = compute_gradient_sum(N, tree, neg_f, status.parallel_buffer);
615 }
616
617 // Compute final t-SNE gradient
618 for (size_t i = 0; i < N * ndim; ++i) {
619 status.dY[i] = status.pos_f[i] - (neg_f[i] / sum_Q);
620 }
621 }

623 template<typename Index>
624 void compute_edge_forces(Status<Index>& status, const Float* Y, Float multiplier) {
625 const auto& neighbors = status.neighbors;
626 auto& pos_f = status.pos_f;
627 std::fill(pos_f.begin(), pos_f.end(), 0);
628
629#ifndef QDTSNE_CUSTOM_PARALLEL
630 #pragma omp parallel for num_threads(nthreads)
631 for (size_t n = 0; n < neighbors.size(); ++n) {
632#else
633 QDTSNE_CUSTOM_PARALLEL(neighbors.size(), [&](size_t first_, size_t last_) -> void {
634 for (size_t n = first_; n < last_; ++n) {
635#endif
636
637 const auto& current = neighbors[n];
638 const Float* self = Y + n * ndim;
639 Float* pos_out = pos_f.data() + n * ndim;
640
641 for (const auto& x : current) {
642 Float sqdist = 0;
643 const Float* neighbor = Y + x.first * ndim;
644 for (int d = 0; d < ndim; ++d) {
645 sqdist += (self[d] - neighbor[d]) * (self[d] - neighbor[d]);
646 }
647
648 const Float mult = multiplier * x.second / (static_cast<Float>(1) + sqdist);
649 for (int d = 0; d < ndim; ++d) {
650 pos_out[d] += mult * (self[d] - neighbor[d]);
651 }
652 }
653
654#ifndef QDTSNE_CUSTOM_PARALLEL
655 }
656#else
657 }
658 }, nthreads);
659#endif
660
661 return;
662 }

719 template<class Algorithm>
720 auto initialize(const Algorithm* searcher) {
721 const int K = perplexity_to_k(perplexity);
722 size_t N = searcher->nobs();
723 if (K >= N) {
724 throw std::runtime_error("number of observations should be greater than 3 * perplexity");
725 }
726
727#ifdef PROGRESS_PRINTER
728 PROGRESS_PRINTER("qdtsne::Tsne::initialize", "Searching for nearest neighbors")
729#endif
730 NeighborList<decltype(searcher->nobs()), Float> neighbors(N);
731
732#ifndef QDTSNE_CUSTOM_PARALLEL
733 #pragma omp parallel for num_threads(nthreads)
734 for (size_t i = 0; i < N; ++i) {
735#else
736 QDTSNE_CUSTOM_PARALLEL(N, [&](size_t first_, size_t last_) -> void {
737 for (size_t i = first_; i < last_; ++i) {
738#endif
739
740 neighbors[i] = searcher->find_nearest_neighbors(i, K);
741
742#ifndef QDTSNE_CUSTOM_PARALLEL
743 }
744#else
745 }
746 }, nthreads);
747#endif
748
749 return initialize_internal(std::move(neighbors), perplexity);
750 }

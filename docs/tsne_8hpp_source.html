468 template<typename Index = int>
469 void run(Status<Index>& status, Float* Y) {
470 int& iter = status.iter;
471 Float multiplier = (iter < stop_lying_iter ? exaggeration_factor : 1);
472 Float momentum = (iter < mom_switch_iter ? start_momentum : final_momentum);
473
474 for(; iter < max_iter; ++iter) {
475 // Stop lying about the P-values after a while, and switch momentum
476 if (iter == stop_lying_iter) {
477 multiplier = 1;
478 }
479 if (iter == mom_switch_iter) {
480 momentum = final_momentum;
481 }
482
483 iterate(status, Y, multiplier, momentum);
484 }
485
486 return;
487 }

496 template<typename Index>
497 void iterate(Status<Index>& status, Float* Y, Float multiplier, Float momentum) {
498 compute_gradient(status, Y, multiplier);
499
500 auto& gains = status.gains;
501 auto& dY = status.dY;
502 auto& uY = status.uY;
503 auto& col_P = status.neighbors;
504
505 // Update gains
506 for (size_t i = 0; i < gains.size(); ++i) {
507 Float& g = gains[i];
508 constexpr Float lower_bound = 0.01;
509 constexpr Float to_add = 0.2;
510 constexpr Float to_mult = 0.8;
511 g = std::max(lower_bound, sign(dY[i]) != sign(uY[i]) ? (g + to_add) : (g * to_mult));
512 }
513
514 // Perform gradient update (with momentum and gains)
515 for (size_t i = 0; i < gains.size(); ++i) {
516 uY[i] = momentum * uY[i] - eta * gains[i] * dY[i];
517 Y[i] += uY[i];
518 }
519
520 // Make solution zero-mean
521 for (int d = 0; d < ndim; ++d) {
522 auto start = Y + d;
523 size_t N = col_P.size();
524
525 // Compute means from column-major coordinates.
526 Float sum = 0;
527 for (size_t i = 0; i < N; ++i, start += ndim) {
528 sum += *start;
529 }
530 sum /= N;
531
532 start = Y + d;
533 for (size_t i = 0; i < N; ++i, start += ndim) {
534 *start -= sum;
535 }
536 }
537
538 return;
539 }

542 template<typename Index>
543 void compute_gradient(Status<Index>& status, const Float* Y, Float multiplier) {
544 auto& tree = status.tree;
545 tree.set(Y);
546
547 compute_edge_forces(status, Y, multiplier);
548
549 size_t N = status.neighbors.size();
550 auto& neg_f = status.neg_f;
551 std::fill(neg_f.begin(), neg_f.end(), 0);
552
553 Float sum_Q = 0;
554 if (interpolation) {
555 sum_Q = interpolate::compute_non_edge_forces(
556 tree,
557 N,
558 Y,
559 theta,
560 neg_f.data(),
561 interpolation
562#ifdef _OPENMP
563 , status.omp_buffer
564#endif
565 );
566
567 } else {
568#ifdef _OPENMP
569 // Don't use reduction methods to ensure that we sum in a consistent order.
570 #pragma omp parallel for
571 for (size_t n = 0; n < N; ++n) {
572 status.omp_buffer[n] = status.tree.compute_non_edge_forces(n, theta, neg_f.data() + n * ndim);
573 }
574 sum_Q = std::accumulate(status.omp_buffer.begin(), status.omp_buffer.end(), static_cast<Float>(0));
575#else
576 for (size_t n = 0; n < N; ++n) {
577 sum_Q += status.tree.compute_non_edge_forces(n, theta, neg_f.data() + n * ndim);
578 }
579#endif
580 }
581
582 // Compute final t-SNE gradient
583 for (size_t i = 0; i < N * ndim; ++i) {
584 status.dY[i] = status.pos_f[i] - (neg_f[i] / sum_Q);
585 }
586 }

588 template<typename Index>
589 void compute_edge_forces(Status<Index>& status, const Float* Y, Float multiplier) {
590 const auto& neighbors = status.neighbors;
591 auto& pos_f = status.pos_f;
592 std::fill(pos_f.begin(), pos_f.end(), 0);
593
594 #pragma omp parallel for
595 for (size_t n = 0; n < neighbors.size(); ++n) {
596 const auto& current = neighbors[n];
597 const Float* self = Y + n * ndim;
598 Float* pos_out = pos_f.data() + n * ndim;
599
600 for (const auto& x : current) {
601 Float sqdist = 0;
602 const Float* neighbor = Y + x.first * ndim;
603 for (int d = 0; d < ndim; ++d) {
604 sqdist += (self[d] - neighbor[d]) * (self[d] - neighbor[d]);
605 }
606
607 const Float mult = multiplier * x.second / (static_cast<Float>(1) + sqdist);
608 for (int d = 0; d < ndim; ++d) {
609 pos_out[d] += mult * (self[d] - neighbor[d]);
610 }
611 }
612 }
613
614 return;
615 }

672 template<class Algorithm>
673 auto initialize(const Algorithm* searcher) {
674 const int K = perplexity_to_k(perplexity);
675 size_t N = searcher->nobs();
676 if (K >= N) {
677 throw std::runtime_error("number of observations should be greater than 3 * perplexity");
678 }
679
680#ifdef PROGRESS_PRINTER
681 PROGRESS_PRINTER("qdtsne::Tsne::initialize", "Searching for nearest neighbors")
682#endif
683 NeighborList<decltype(searcher->nobs()), Float> neighbors(N);
684
685 #pragma omp parallel for
686 for (size_t i = 0; i < N; ++i) {
687 neighbors[i] = searcher->find_nearest_neighbors(i, K);
688 }
689
690 return initialize_internal(std::move(neighbors), perplexity);
691 }

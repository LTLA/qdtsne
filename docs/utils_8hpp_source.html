56template<int num_dim_, typename Float_ = double>
57void initialize_random(Float_* Y, size_t num_points, int seed = 42) {
58 std::mt19937_64 rng(seed);
59
60 size_t total = num_points * num_dim_;
61 bool odd = total % 2;
62 if (odd) {
63 --total;
64 }
65
66 // Box-Muller gives us two random values at a time.
67 for (size_t i = 0; i < total; i += 2) {
68 auto paired = aarand::standard_normal<Float_>(rng);
69 Y[i] = paired.first;
70 Y[i + 1] = paired.second;
71 }
72
73 if (odd) {
74 // Adding the poor extra for odd total lengths.
75 auto paired = aarand::standard_normal<Float_>(rng);
76 Y[total] = paired.first;
77 }
78
79 return;
80}
